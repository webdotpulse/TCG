Dialogs & Modals

How to use:

The modal system uses a full-screen overlay with a centered dialog box. Add the following structure to the end of your HTML `` to ensure it overlays everything.

<!-- Interactive Modal (Hidden by default) -->
<div id="demo-modal" class="fixed inset-0 z-100 hidden">
    <!-- Backdrop -->
    <div class="absolute inset-0 bg-slate-900/50 backdrop-blur-sm transition-opacity"></div>

    <!-- Modal Content Wrapper -->
    <div class="fixed inset-0 z-10 overflow-y-auto">
        <div class="flex min-h-full items-center justify-center p-4 text-center sm:p-0">
            <!-- Modal Dialog -->
            <div class="relative transform overflow-hidden rounded-2xl bg-white text-left shadow-2xl transition-all sm:my-8 sm:w-full sm:max-w-lg scale-95 opacity-0" id="demo-modal-dialog">
                <!-- Header -->
                <div class="flex items-center justify-between p-5 border-b border-slate-100">
                    <h3 class="text-lg font-bold text-slate-900">Interactive Modal</h3>
                    <button class="close-modal text-slate-400 hover:text-slate-700 hover:bg-slate-100 rounded-lg p-1.5 transition-colors">
                        <i data-lucide="x"></i>
                    </button>
                </div>
                <!-- Body -->
                <div class="p-6">
                    <p class="text-slate-600 text-base leading-relaxed">This modal is fully interactive and controlled via JavaScript. You can use it to display important information, forms, or confirmations.</p>
                </div>
                <!-- Footer -->
                <div class="flex items-center justify-end gap-3 p-5 border-t border-slate-100 bg-slate-50">
                    <button class="close-modal px-4 py-2 text-base font-medium text-slate-700 bg-white border border-slate-300 rounded-lg hover:bg-slate-50 transition-colors">Cancel</button>
                    <button class="close-modal px-4 py-2 text-base font-medium text-white bg-brand-blue rounded-lg hover:bg-brand-teal transition-colors shadow-sm">Confirm Action</button>
                </div>
            </div>
        </div>
    </div>
</div>

Toasts

How to use:

Toasts are used to show temporary, non-disruptive notifications. Add a container fixed to a corner (like `fixed bottom-4 right-4`) and then add toast elements inside it.

<!-- Toast Container (bottom right) -->
<div id="toast-container" class="fixed bottom-4 right-4 z-110 flex flex-col gap-3">
    <!-- A Toast element (hidden initially or generated via JS) -->
    <div id="demo-toast" class="bg-white border border-slate-200 shadow-xl rounded-xl p-4 w-80 transform transition-all translate-x-full opacity-0 flex items-start gap-3">
        <div class="text-green-500 mt-0.5">
            <i data-lucide="check-circle"></i>
        </div>
        <div class="flex-1">
            <h4 class="text-base font-bold text-slate-900">Action Successful</h4>
            <p class="text-sm text-slate-500 mt-1">Your changes have been saved successfully.</p>
        </div>
        <button class="close-toast text-slate-400 hover:text-slate-700 transition-colors">
            <i data-lucide="x"></i>
        </button>
    </div>
</div>
